# AutoPilot 环境变量模板
# 请将以下内容复制到青龙面板的环境变量中，并填写实际值

# 京东配置
JD_COOKIE=pt_key=AAAA;pt_pin=BBB;

# 通知配置
PUSH_PLUS_TOKEN=你的PushPlusToken
TG_BOT_TOKEN=你的TelegramBotToken
TG_USER_ID=你的用户ID

# 其他平台配置
BILIBILI_COOKIE=你的B站Cookie
ALIYUNDRIVE_TOKEN=你的阿里云盘Token

# 服务器监控配置
SERVER_MONITOR_URL=你的服务器监控地址
SSL_DOMAINS=需要监控的域名,多个用逗号分隔
